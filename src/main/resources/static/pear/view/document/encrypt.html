API 文档
encrypt.md5( str ) -- MD5加密

encrypt.sha1( str ) -- SHA1加密

encrypt.sha256( str ) -- SHA256加密

encrypt.sha512( str ) -- SHA512加密

encrypt.rmd160( str ) -- RMD160加密

encrypt.crc32( str ) -- CRC32加密

encrypt.Base64Encode( str ) -- BASE64加密
